110 static inline uint16_t FAT_DATE(uint16_t year, uint8_t month, uint8_t day) {
111  return (year - 1980) << 9 | month << 5 | day;
112 }
118 static inline uint16_t FAT_YEAR(uint16_t fatDate) {
